\subheading{Recall}

\begin{text-box}
    A collection $\tau$ of subsets of $X$ is called a topology on $X$ if \\
    (i) $X, \emptyset \in \tau$. \\
    (ii) The union of any collection of sets in $\tau$ is also in $\tau$. \\
    (iii) The finite intersection of sets in $\tau$ is also in $\tau$.
\end{text-box}

\begin{text-box}
    Let $(X, \tau)$: a topological space. \\
    $V \in \tau$이면 $V$: open set in $X$.    
\end{text-box}
